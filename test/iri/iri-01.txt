:Resource 
    a owl:Class ;
    rdfs:label "Resource" ;
    rdfs:isDefinedBy <http://purl.jp/bio/11/orth> .

:Dataset 
    a owl:Class ;
    rdfs:label "Dataset" ;
    rdfs:subClassOf [
        a owl:Restriction ;
	owl:onProperty :organism ;
	owl:allValuesFrom :Organism 
    ] ;
    rdfs:subClassOf :Resource ;
    rdfs:subClassOf void:Dataset ;
    owl:equivalentClass orthology:OrthologyDataset ;
    rdfs:isDefinedBy <http://purl.jp/bio/11/orth> .

void:Dataset 
    rdfs:subClassOf dctype:Dataset .

:Group 
    a owl:Class ;
    rdfs:label "Group" ;
    rdfs:subClassOf [
        a owl:Restriction ;
	owl:onProperty :inDataset ;
	owl:allValuesFrom :Dataset 
    ] ;
    rdfs:subClassOf [
        a owl:Restriction ;
	owl:onProperty :member ;
	owl:allValuesFrom [
	    a owl:Class ;
	    owl:unionOf ( :Group 
	        :SequenceUnit 
            )
        ] ;
    ] ;
    rdfs:subClassOf :Resource ;
    rdfs:subClassOf dctype:Collection ;
    rdfs:isDefinedBy <http://purl.jp/bio/11/orth> .
